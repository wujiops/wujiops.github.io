nginx ssl

# nginx yum源配置
[root@nginx conf.d]# cat /etc/yum.repos.d/nginx.repo 
[nginx-stable]
name=nginx stable repo
baseurl=http://nginx.org/packages/centos/$releasever/$basearch/
gpgcheck=1
enabled=1
gpgkey=https://nginx.org/keys/nginx_signing.key
module_hotfixes=true
# 配置文件
[root@nginx conf.d]# cat gitlab.conf
# 80 -- > 443 自动跳转
server {
    listen       80;
    server_name  gitlab.wujiops.cn;
    return 301 https://$server_name$request_uri;
}
server {
    listen       443 ssl;
    server_name  gitlab.wujiops.cn;
    charset utf-8;
    # 证书配置
    ssl_certificate /etc/nginx/cert/wujiops.cn.cer;
    ssl_certificate_key /etc/nginx/cert/wujiops.cn.key;
    ssl_session_timeout 1d;
    ssl_ciphers EECDH+CHACHA20:EECDH+CHACHA20-draft:EECDH+AES128:RSA+AES128:EECDH+AES256:RSA+AES256::!MD5;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_session_cache shared:MozSSL:10m;
    ssl_prefer_server_ciphers on;
    add_header Strict-Transport-Security "max-age=63072000" always;
    client_max_body_size 4096m;
    access_log /home/app/logs/gitlab_access.log  main;
    error_log  /home/app/logs/gitlab_error.log;
    # nginx 反向代理配置
    location / {
        proxy_redirect http:// https://;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-Ssl on;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_pass http://192.168.1.15:18080;
    }

    error_page 404 /404.html;
    location = /404.html {
    }

    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
    }
}

apache ssl

检查apache是否启用mod_ssl模块，一般情况下会在/etc/httpd/conf.d/下存在ssl.conf文件。

依赖安装

安装mod_ssl模块

yum install -y mod_ssl

修改ssl.conf文件

<VirtualHost _default_:443>
DocumentRoot "/var/www/html"
ServerName roxn.cn:443
SSLEngine on
SSLProtocol all -SSLv2 -SSLv3
SSLCipherSuite HIGH:!RC4:!MD5:!aNULL:!eNULL:!NULL:!DH:!EDH:!EXP:+MEDIUM
SSLHonorCipherOrder on
SSLCipherSuite PROFILE=SYSTEM
SSLProxyCipherSuite PROFILE=SYSTEM
SSLCertificateFile /etc/pki/tls/certs/apache/4434998_roxn.cn_public.crt
SSLCertificateKeyFile /etc/pki/tls/certs/apache/4434998_roxn.cn.key

修改httpd.conf，使http跳转到https

<VirtualHost *:80>

    RewriteEngine on
    RewriteCond %{SERVER_PORT} !^443$
    RewriteRule ^(.*)$ https://%{SERVER_NAME}$1 [L,R]

</VirtualHost>

重启验证

systemctl restart httpd
