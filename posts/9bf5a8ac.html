#!/bin/bash
# 安装密钥生成工具和批量处理工具
yum -y install expect sshpass
# 无交互生成密钥
spawn ssh-keygen
expect {
        "*(~/.ssh/id_rsa)" {send "\r";exp_continue}
        "*(empty for no passphrase)" {send "\r";exp_continue}
        "*again" {send "\r"}
}
expect eof
# 批量分发
#for ip in $*
for ip in $(seq 101 111)

do
  ping 192.168.1.$ip -c1 &>/dev/null 
  if [ $? -gt 0 ];then
    echo "192.168.1.$ip无法ping通请检查网络" 
    continue
  fi
  sshpass -p "pass@word1" ssh-copy-id -o StrictHostKeyChecking=no root@192.168.1.$ip &>/dev/null
  echo "192.168.1.$ip 密钥分发成功"
done

################################
#1.使用ssh连接远程主机时加上"-o StrictHostKeyChecking=no"的选项，如下：（推荐！！！）

#scp -o "StrictHostKeyChecking no" 1.txt user@host:1.txt 
#ssh -o "StrictHostKeyChecking no" user@host
#2.一个彻底去掉这个提示的方法是，修改当前主机/etc/ssh/ssh_config文件（或$HOME/.ssh/config）中的配置，添加如下两行配置：
#源为：StrictHostKeyChecking ask
#改为：StrictHostKeyChecking no

安装系统,优化脚本

一定要修改脚本,为自己可用

将脚本分发给所有主机并执行.

#!/bin/bash

#Yum源更换为国内阿里源
yum install wget telnet -y
cd /etc/yum.repos.d/
mkdir repo_old
mv *.repo repo_old/
wget -O /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-7.repo

#添加阿里的epel源
wget -O /etc/yum.repos.d/epel.repo http://mirrors.aliyun.com/repo/epel-7.repo

#添加ambari本地源
cat >> /etc/yum.repos.d/ambari.repo << EOF
[ambari]
name=ambari
baseurl=http://192.168.1.200/ambari/centos7/2.7.3.0-139/
gpgcheck=0
enabled=1

[HDP]
name=HDP
baseurl=http://192.168.1.200/HDP/centos7/3.1.0.0-78/
gpgcheck=0
enabled=1

[HDP-UTILS]
name=HDP-UTILS
baseurl=http://192.168.1.200/HDP-UTILS/centos7/1.1.0.22/
gpgcheck=0
enabled=1
EOF

#修改hosts文件
cat >> /etc/hosts << EOF
192.168.1.101  master1.hadoop.com
192.168.1.102  master2.hadoop.com
192.168.1.103  master3.hadoop.com
192.168.1.104  slave1.hadoop.com
192.168.1.105  slave2.hadoop.com
192.168.1.108  client.hadoop.com
192.168.1.110  ambari.hadoop.com
EOF

#yum重新建立缓存
yum clean all
yum makecache
#同步时间
yum -y install ntp
/usr/sbin/ntpdate cn.pool.ntp.org
echo "* 4 * * * /usr/sbin/ntpdate cn.pool.ntp.org > /dev/null 2>&1" >> /var/spool/cron/root
systemctl  restart crond.service

#安装vim
yum -y install vim

#设置最大打开文件描述符数
echo "ulimit -SHn 102400" >> /etc/rc.local
cat >> /etc/security/limits.conf << EOF
*           soft   nofile       655350
*           hard   nofile       655350
EOF

#禁用selinux & firewalld
sed -i 's/SELINUX=enforcing/SELINUX=disabled/' /etc/selinux/config
setenforce 0

systemctl disable firewalld.service 
systemctl stop firewalld.service 

#ssh设置
#sed -i 's/^GSSAPIAuthentication yes$/GSSAPIAuthentication no/' /etc/ssh/sshd_config
#sed -i 's/#UseDNS yes/UseDNS no/' /etc/ssh/sshd_config
#systemctl  restart sshd.service

#内核参数优化
