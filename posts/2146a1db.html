配置Ubuntu docker-ce 阿里云apt源

wang@roxn:/etc/apt$ sudo apt-get update   # 更新apt源
wang@roxn:/etc/apt$ sudo apt-get -y install apt-transport-https ca-certificates curl software-properties-common   # 安装系统工具
wang@roxn:/etc/apt$ curl -fsSL http://mirrors.aliyun.com/docker-ce/linux/ubuntu/gpg | sudo apt-key add -   # 添加GPG证书
wang@roxn:/etc/apt$ sudo add-apt-repository "deb [arch=amd64] http://mirrors.aliyun.com/docker-ce/linux/ubuntu $(lsb_release -cs) stable"   # 设置Ubuntu的docker-ce的阿里云apt源
wang@roxn:/etc/apt$ sudo apt-get -y update   # 更新apt源
wang@roxn:/etc/apt$ sudo apt-get -y install docker-ce   # 安装docker-ce

centos 7 安装docker-ce

关闭firewalld和selinux

systemctl stop firewalld
systemctl disable firewalld
setenforce 0
sed -i 's/enforcing$/disabled/g' /etc/selinux/config

安装依赖

yum install -y yum-utils device-mapper-persistent-data lvm2

增加aliyun dockeryum源

yum-config-manager --add-repo https://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo

安装启动

yum makecache fast   # 建立缓存
yum -y install docker-ce
service docker start

docker pull加速

mkdir -p /etc/docker
tee /etc/docker/daemon.json <<-'EOF'
{
  "registry-mirrors": ["https://fogd2pyo.mirror.aliyuncs.com"]
}
EOF

重启docker

systemctl daemon-reload   # 加载配置
systemctl restart docker
