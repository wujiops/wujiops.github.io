安装openvpn

[root@wangj openvpn]# cat /etc/redhat-release 
Anolis OS release 7.9
# 安装epel-release源，yum直接安装openvpn。
yum install openssh-server openssl openssl-devel lzo lzo-devel pam-devel openvpn -y

目录

/etc/openvpn

配置证书

cd /etc/openvpn
# 安装生成证书工具
yum install easy-rsa -y
cp -a /usr/share/easy-rsa/3.0.8 /etc/openvpn/easyrsa
cd /etc/openvpn/easyrsa
# 生成证书配置文件，tee直接覆盖，tee -a为追加内容
tee vars <<- 'EOF'
set_var EASYRSA_DN "cn_only"
set_var EASYRSA_REQ_COUNTRY "CN"
set_var EASYRSA_REQ_PROVINCE "Beijing"
set_var EASYRSA_REQ_CITY "Beijing"
set_var EASYRSA_REQ_ORG "roxn"
set_var EASYRSA_REQ_EMAIL "my@roxn.cn"
set_var EASYRSA_NS_SUPPORT "yes"
set_var EASYRSA_CA_EXPIRE 36135
set_var EASYRSA_KEY_EXPIRE 36135
EOF
# set_var EASYRSA_CA_EXPIRE 36135   # 过期时间
# set_var EASYRSA_KEY_EXPIRE 36135   # 按实际要求修改
# 初始化与创建CA根证书
./easyrsa init-pki
# 初始化，会在当前目录创建PKI目录，用于存储一些中间变量及最终生成的证书
./easyrsa build-ca
# 在这部分需要输入PEM密码 PEM pass phrase，输入两次，此密码必须记住，不然以后不能为证书签名。
# 还需要输入common name 通用名，如：openvpn，这个你自己随便设置个独一无二的。

# 生成服务端证书
./easyrsa build-server-full server nopass
# 为服务端生成证书对并在本地签名。nopass参数生成一个无密码的证书；在此过程中会让你确认ca密码

./easyrsa gen-dh
# 创建Diffie-Hellman，确保key穿越不安全网络的命令，时间会有点长，耐心等待

# 为了提高安全性，生成ta.key
openvpn --genkey --secret ta.key
# 加强认证方式，防攻击。

# 整理服务端证书
cp -a pki/ca.crt /etc/openvpn/server/
cp -a pki/private/server.key /etc/openvpn/server/
cp -a pki/issued/server.crt /etc/openvpn/server/
cp -a pki/dh.pem /etc/openvpn/server/
cp -a ta.key /etc/openvpn/server/

服务端配置文件

# 创建服务端配置文件
tee /etc/openvpn/server/server.conf <<- 'EOF'
#监控本机端口号
port 52987

#指定采用的传输协议，可以选择tcp或udp
proto tcp

#指定创建的通信隧道类型，可选tun或tap
dev tun

#指定CA证书的文件路径
ca /etc/openvpn/easyrsa/pki/ca.crt

#指定服务器端的证书文件路径
cert /etc/openvpn/easyrsa/pki/issued/server.crt

#指定服务器端的私钥文件路径
key /etc/openvpn/easyrsa/pki/private/server.key

#指定迪菲赫尔曼参数的文件路径
dh /etc/openvpn/easyrsa/pki/dh.pem

#指定虚拟局域网占用的IP地址段和子网掩码，此处配置的服务器自身占用.1的ip地址
server 10.0.0.0 255.255.255.0

#推送路由到客户端，实现VPN客户端之间通讯
#push "route 0.0.0.0 0.0.0.0"
push "route 10.0.0.0 255.255.255.0"
#推送路由到客户端，本机网段172.21.0.0，实现VPN客户端和本机网段内的内网主机通讯
#实际测试不需要在server端使用iptables转发
push "route 172.21.0.0 255.255.255.0"

#服务器自动给客户端分配IP后，客户端下次连接时，仍然采用上次的IP地址(第一次分配的IP保存在ipp.txt中，下一次分配其中保存的IP)。
ifconfig-pool-persist ipp.txt

#自动推送客户端上的网关及DHCP
#push "redirect-gateway def1 bypass-dhcp"

#OpenVPN的DHCP功能为客户端提供指定的 DNS、WINS 等
#push "dhcp-option DNS 114.114.114.114"

#允许客户端与客户端相连接，默认情况下客户端只能与服务器相连接
client-to-client

#每10秒ping一次，连接超时时间设为120秒
keepalive 10 120

#开启TLS-auth，使用ta.key防御攻击。服务器端的第二个参数值为0，客户端的为1。
tls-auth /etc/openvpn/easyrsa/ta.key 0

#加密认证算法
cipher AES-256-CBC

#使用lzo压缩的通讯,服务端和客户端都必须配置
comp-lzo

#最大连接用户
max-clients 100 

#重启时仍保留一些状态
persist-key
persist-tun

#输出短日志,每分钟刷新一次,以显示当前的客户端
status /home/logs/vpn/openvpn-status.log

#日志保存路径
log         /home/logs/vpn/openvpn.log
log-append  /home/logs/vpn/openvpn.log

#指定日志文件的记录详细级别，可选0-9，等级越高日志内容越详细
verb 4

#相同信息的数量，如果连续出现 20 条相同的信息，将不记录到日志中
mute 20

#注销证书
crl-verify /etc/openvpn/easyrsa/pki/crl.pem
EOF
mkdir -p /home/logs/vpn/
systemctl start openvpn-server@server

快捷脚本

openvpn

#!/bin/bash

case $1 in
add)
  ./easyrsa gen-req $2 nopass << EOF

EOF
  ./easyrsa sign client $2 << EOF
yes
EOF
  ls -l pki/private/${2}.key pki/issued/${2}.crt
;;
del)
  ./easyrsa revoke ${2} << EOF
  rm -f pki/private/${2}.key pki/issued/${2}.crt
yes
EOF
  rm -f pki/private/${2}.key pki/issued/${2}.crt
./easyrsa gen-crl >> /dev/null 2>&1
;;
cp)
  rm -f cccp/*
  cp none/client.default cccp/client.default
  echo "<cert>" >> cccp/client.default
  cat pki/issued/${2}.crt >> cccp/client.default
  echo "</cert>" >> cccp/client.default
  echo "<key>" >> cccp/client.default
  cat pki/private/${2}.key >> cccp/client.default
  echo "</key>">> cccp/client.default
  cp cccp/client.{default,conf}
  mv cccp/client.{default,ovpn}
;;
scp)
  rm -f cccp/*
  cp none/client.default cccp/client.default
  echo "<cert>" >> cccp/client.default
  cat pki/issued/${2}.crt >> cccp/client.default
  echo "</cert>" >> cccp/client.default
  echo "<key>" >> cccp/client.default
  cat pki/private/${2}.key >> cccp/client.default
  echo "</key>">> cccp/client.default
  cp cccp/client.{default,conf}
  mv cccp/client.{default,ovpn}
  ssh -i /home/wangj/.ssh/id_rsa -p32601 roxn@jump "rm -f /home/roxn/vpn/*"
  scp -i /home/wangj/.ssh/id_rsa -P 32601 cccp/* roxn@jump:vpn/
;;
*)
  echo "vpn add|del|cp|scp name"
;;
esac

脚本使用方法

vpn add|del|cp|scp name
add   增加用户证书
del   吊销证书
cp    复制证书到cccp目录，生成client.conf/ovpn两个配置文件，分别可以在Linux和Windows使用
scp   我这里设定的是发送到远端固定服务器

结语

安装部署可能会有一些小细节未完善，百度就能解决，常见的就是启动报错，一般是日志路径or日志文件不存在，手动创建即可，还有就是吊销证书报错，一般也是报错文件不存在，最好是创建xxx用户先吊销一次，后续再创建用户证书就可以直接吊销了。还有使用用户名密码连接openvpn的，这里不再做赘述，自行百度即可。还有openvpn server端是可以同时启动多个的，指定不同的配置文件即可，可以参考ps -ef|grep openvpn的信息配置，这样可以实现两个openvpn server端同时运行，公共使用的vpn server可以在夜间关闭。另外centos8及其分支系统，dnf安装openvpn server端可能不会生成/usr/lib/systemd/system/openvpn@.service文件，使用下面的脚本生成即可。

cat >> /usr/lib/systemd/system/openvpn@.service << EOF
[Unit]
Description=OpenVPN Robust And Highly Flexible Tunneling Application On %I
After=network.target

[Service]
Type=notify
PrivateTmp=true
ExecStart=/usr/sbin/openvpn --cd /etc/openvpn/ --config %i.conf

[Install]
WantedBy=multi-user.target
EOF
