#!/usr/bin/env bash
# @File    :   init.sh
# @Time    :   2021/12/06 15:41:17
# @Author  :   Roxn
# @Email   :   158970251@qq.com
# @Software:   VS Code

# 日志存放路径
log="/home/logs/init.log"

if [ ! -e $log ]; then
  mkdir -p $(echo $log | awk -v FS=/ -v OFS=/ '{$NF=""; print $0}')
  touch $log
fi

# 如果执行过程中有错误信息均输出到日志文件中
# ex: 2>&1
fsize=2000000
exec >> $log

# 日志函数
# 参数一，级别: INFO WARN ERROR

function LOGS() {

  # local curtime
  curtime=$(date +"%F %T")
  timestamp=$(date +%F_%H_%M_%S)

  # local cursize
  cursize=$(cat $log | wc -c)

  if [ $fsize -lt $cursize ]; then
    mv $log{,.$timestamp}
    touch $log
  fi

}
function INFO() {
  LOGS
  echo -e "$curtime \033[32m[INFO]\033[0m $*" >> $log
}
function WARN() {
  LOGS
  echo -e "$curtime \033[33m[WARN]\033[0m $*" >> $log
}
function ERROR() {
  LOGS
  echo -e "$curtime \033[31m[ERROR]\033[0m $*" >> $log
}

workdir=$(cd $(dirname $0); pwd)

<<'EOF'
check init file 禁止2次初始化操作
check os 系统版本检查
7-8 对类centos7-8的国产Linux系统初始化(opencloudos8 anolins7-8 centos7-8)，未在openeuler和9系统上面测试。
原理上可以在Alibaba Cloud Linux系统上运行，但是未做测试。
EOF

INFO "checking /etc/.init"
# [[ -e ~/.init ]] && printf "true\n" || printf "false\n"
if [[ -e /.init ]]; then
  WARN "line $LINENO. Warnning inited."
  printf "line $LINENO. Warnning inited.\n"
  exit 999
else
  INFO "check OK"
  printf "$curtime inited\n" > /.init
fi

INFO "chekck /etc/os-release"
SysList="CentOS Anolis OpenCloudOS"

OS=$(head -1 /etc/os-release)

# [[ "$OS" =~ "Anolis" ]] || [[ "$OS" =~ "CentOS" ]] && printf "pass\n" || printf "error\n"

if [[ "$OS" =~ "Anolis" ]] || [[ "$OS" =~ "CentOS" ]] || [[ "$OS" =~ "OpenCloudOS" ]]; then
  INFO "chekck /etc/os-release pass"
else
  WARN "line $LINENO. Warnning check os err."
  printf "line $LINENO. Warnning check os err.\n"
  exit 998
fi

Version=$(grep ^VERSION= /etc/os-release | grep -Eo [0-9]+ | cut -d. -f1)
SysVersion=$(grep PRETTY_NAME= /etc/os-release | awk -F\" '{print $2}')

if [[ $Version != 7 ]] && [[ $Version != 8 ]]; then
    WARN "line $LINENO. check OS error."
    printf "line $LINENO. check OS error."
    exit 997
else
    INFO "$SysVersion check pass"
fi

INFO "install package && update"
if test $Version = 7; then
  INFO "yum -y install lrzsz vim wget epel-release ntp unzip bash-completion"
  yum -y install lrzsz vim wget epel-release ntp unzip bash-completion >> $log 2>&1
  INFO "yum update -y"
  yum update -y >> $log 2>&1
elif test $Version = 8; then
  INFO "rpm -ivh http://mirrors.wlnmp.com/centos/wlnmp-release-centos.noarch.rpm"
  rpm -ivh http://mirrors.wlnmp.com/centos/wlnmp-release-centos.noarch.rpm >> $log 2>&1
  INFO "dnf -y install lrzsz vim wget epel-release glibc-langpack-en wntp bash-completion"
  dnf -y install lrzsz vim wget epel-release glibc-langpack-en wntp bash-completion >> $log 2>&1
  INFO "dnf update -y"
  dnf update -y >> $log 2>&1
fi

INFO "close selinux & firewald"

INFO "set /etc/selinux/config SELINUX=disabled"
sed -i.bak 's/SELINUX=enforcing/SELINUX=disabled/g' /etc/selinux/config
setenforce 0
INFO "systemctl disable firewalld"
systemctl stop firewalld
systemctl disable firewalld >> $log 2>&1

INFO "set sshd & vim & profile"

INFO "set /etc/ssh/sshd_config UseDNS no"
if test $Version = 7; then
  sed -i.bak '/#UseDNS/a UseDNS\ no' /etc/ssh/sshd_config
elif test $Version = 8; then
  sed -i.bak 's/^#\(UseDNS no\)/\1/' /etc/ssh/sshd_config
fi

systemctl restart sshd

INFO "alias vi=vim & tabstop & F9"
echo 'alias vi=vim' >> /etc/profile

cat >> /root/.vimrc << EOF
set tabstop=4
set shiftwidth=4
set expandtab
set number
set pastetoggle=<F9>
set paste
EOF

INFO "set sysctl.conf"
if test $Version = 7; then
  cat >> /etc/sysctl.conf << EOF
# 开启SYN Cookies，当SYN等待队列溢出时，启用cookies来处理，可以防范少量的SYN攻击，默认为0，表示关闭
net.ipv4.tcp_syncookies = 1
# 允许将TIME_WAIT sockets重新用于新的TCP连接，默认为0，表示关闭
net.ipv4.tcp_tw_reuse = 1
# 允许将TIME_WAIT sockets快速回收以便利用，默认为0，表示关闭，需要同时开启 net.ipv4.tcp_timestamps 才能生效
net.ipv4.tcp_tw_recycle = 1
net.ipv4.tcp_timestamps = 1
# 设置TCP三次请求的fin状态超时
net.ipv4.tcp_fin_timeout = 30
# 设置TCP 发送keepalive的频度，默认的缺省为2小时，600秒表示10分钟，表示服务器以10分钟发送keepalive消息
net.ipv4.tcp_keepalive_time = 600
# 探测包发送的时间间隔设置为2秒，默认75秒
net.ipv4.tcp_keepalive_intvl = 2
# 如果对方不给予应答，探测包发送的次数，默认9次
net.ipv4.tcp_keepalive_probes = 2
# 设置本地端口范围，缺省情况下: 32768到61000，现在改为 1024 到 65000，最小值不能设置太低,否则占用了正常端口
net.ipv4.ip_local_port_range = 1024 65000
# 表示SYN队列的长度，默认为1024，加大队列长度为8192，可以容纳更多的网络连接数
net.ipv4.tcp_max_syn_backlog = 8192
# 设置保持TIME_WAIT的最大数量，如果超过这个数量，TIME_WAIT将立刻清楚并打印警告信息，默认为180000，改为5000.此项参数可以控制TIME_WAIT的最大数量,避免Squid服务器被大量的TIME_WAIT拖死
net.ipv4.tcp_max_tw_buckets = 5000
# 表示SYN队列的长度，选项为服务器端用于记录那些尚未收到客户端确认信息的连接请求的最大值，该参数对应系统路径为: /proc/sys/net/ipv4/tcp_max_syn_backlog
net.ipv4.tcp_max_syn_backlog = 4096
# 增加tcp缓冲区大小，tcp_rmem表示接受数据缓冲区范围，tcp_wmem表示发送数据缓冲区范围，单位Byte，最大64M
net.ipv4.tcp_rmem = 4096 87380 67108864
net.ipv4.tcp_wmem = 4096 65536 67108864
# TCP失败重传次数,默认值15，意味着重传15次才彻底放弃，可减少到5，以尽早释放内核资源
net.ipv4.tcp_retries2 = 5
# 选项默认值是128，这个参数用于调节系统同时发起的tcp连接数，在高并发请求中，默认的值可能会导致连接超时或重传，因此，需要结合并发请求数来调节此值。该参数对应系统路径为: /proc/sys/net/core/somaxconn 128
net.core.somaxconn = 4096
# 启用timewait 快速回收
net.ipv4.tcp_tw_recycle = 1
EOF
elif test $Version = 8; then
  cat >> /etc/sysctl.conf << EOF
# 开启SYN Cookies，当SYN等待队列溢出时，启用cookies来处理，可以防范少量的SYN攻击，默认为0，表示关闭
net.ipv4.tcp_syncookies = 1
# 允许将TIME_WAIT sockets重新用于新的TCP连接，默认为0，表示关闭
net.ipv4.tcp_tw_reuse = 1
net.ipv4.tcp_timestamps = 1
# 设置TCP三次请求的fin状态超时
net.ipv4.tcp_fin_timeout = 30
# 设置TCP 发送keepalive的频度，默认的缺省为2小时，600秒表示10分钟，表示服务器以10分钟发送keepalive消息
net.ipv4.tcp_keepalive_time = 600
# 探测包发送的时间间隔设置为2秒，默认75秒
net.ipv4.tcp_keepalive_intvl = 2
# 如果对方不给予应答，探测包发送的次数，默认9次
net.ipv4.tcp_keepalive_probes = 2
# 设置本地端口范围，缺省情况下: 32768到61000，现在改为 1024 到 65000，最小值不能设置太低,否则占用了正常端口
net.ipv4.ip_local_port_range = 1024 65000
# 表示SYN队列的长度，默认为1024，加大队列长度为8192，可以容纳更多的网络连接数
net.ipv4.tcp_max_syn_backlog = 8192
# 设置保持TIME_WAIT的最大数量，如果超过这个数量，TIME_WAIT将立刻清楚并打印警告信息，默认为180000，改为5000.此项参数可以控制TIME_WAIT的最大数量,避免Squid服务器被大量的TIME_WAIT拖死
net.ipv4.tcp_max_tw_buckets = 5000
# 表示SYN队列的长度，选项为服务器端用于记录那些尚未收到客户端确认信息的连接请求的最大值，该参数对应系统路径为: /proc/sys/net/ipv4/tcp_max_syn_backlog
net.ipv4.tcp_max_syn_backlog = 4096
# 增加tcp缓冲区大小，tcp_rmem表示接受数据缓冲区范围，tcp_wmem表示发送数据缓冲区范围，单位Byte，最大64M
net.ipv4.tcp_rmem = 4096 87380 67108864
net.ipv4.tcp_wmem = 4096 65536 67108864
# TCP失败重传次数,默认值15，意味着重传15次才彻底放弃，可减少到5，以尽早释放内核资源
net.ipv4.tcp_retries2 = 5
# 选项默认值是128，这个参数用于调节系统同时发起的tcp连接数，在高并发请求中，默认的值可能会导致连接超时或重传，因此，需要结合并发请求数来调节此值。该参数对应系统路径为: /proc/sys/net/core/somaxconn 128
net.core.somaxconn = 4096
# 设置tcp确认超时时间 300秒，默认 432000 秒（5天）
net.netfilter.nf_conntrack_tcp_timeout_established = 300
# 设置tcp等待时间 12秒，超过12秒自动放弃，默认120秒
net.netfilter.nf_conntrack_tcp_timeout_time_wait = 12
# 设置tcp关闭等待时间60秒，超过60秒自动关闭，默认60秒
net.netfilter.nf_conntrack_tcp_timeout_close_wait = 60
# 设置tcp fin状态的超时时间为120秒,超过该时间自动关闭，默认120秒
net.netfilter.nf_conntrack_tcp_timeout_fin_wait = 120
EOF
fi
sysctl -p

INFO "set ntp server"
if test $Version = 7; then
  /usr/sbin/ntpdate ntp.ntsc.ac.cn
  crontab -l > /tmp/.txt 2>&1
  echo "*/30 * * * * /usr/sbin/ntpdate ntp.ntsc.ac.cn > /dev/null 2>&1" >> /tmp/.txt
  crontab /tmp/.txt
  echo "alias ntp='/usr/sbin/ntpdate ntp.ntsc.ac.cn'" >> /etc/profile.d/ntp.sh
  systemctl enable ntpd >> $log 2>&1
  systemctl start ntpd
  systemctl restart crond.service
elif test $Version = 8; then
  crontab -l > /tmp/.txt 2>&1
  echo "*/30 * * * * /usr/local/bin/ntpdate ntp.ntsc.ac.cn > /dev/null 2>&1" >> /tmp/.txt
  crontab /tmp/.txt
  systemctl restart crond.service
fi

INFO "set timezone"
timedatectl set-timezone Asia/Shanghai

INFO "set hwclock"
/usr/sbin/hwclock --systohc
/usr/sbin/hwclock -w

INFO "set limits.conf"
cat >> /etc/security/limits.conf << EOF
* soft nofile 190000
* hard nofile 200000
* soft nproc 252144
* hadr nproc 262144
EOF

INFO "set history"
cat >> /etc/bashrc << EOF
HISTFILESIZE=10000000
HISTSIZE=200
HISTTIMEFORMAT='%F %T '
export HISTTIMEFORMAT
EOF

INFO "unset MAILCHECK"
cat >> /etc/profile << EOF
HISTSIZE=200
PROMPT_COMMAND="history -a"
shopt -s histappend
unset MAILCHECK
EOF

source /etc/profile

INFO "set /etc/hosts"
cat >> /etc/hosts<<EOF
192.168.1.30 nnv nas
192.168.1.31 node1
192.168.1.32 node2
192.168.1.33 node3
192.168.1.35 gitlab gitlab.roxn.cn
EOF

INFO "set /usr/lib/systemd/system/openvpn@.service"
if test $Version = 8; then
  cat >> /usr/lib/systemd/system/openvpn@.service << EOF
[Unit]
Description=OpenVPN Robust And Highly Flexible Tunneling Application On %I
After=network.target

[Service]
Type=notify
PrivateTmp=true
ExecStart=/usr/sbin/openvpn --cd /etc/openvpn/ --config %i.conf

[Install]
WantedBy=multi-user.target
EOF
fi

reboot
