环境

Anolis OS 8.8
4C8G
java 8

安装

官方文档

wget -O /etc/yum.repos.d/jenkins.repo https://pkg.jenkins.io/redhat-stable/jenkins.repo
rpm --import https://pkg.jenkins.io/redhat-stable/jenkins.io-2023.key
dnf install fontconfig java-11-openjdk -y
# 默认安装jenkins-lts
dnf install jenkins -y

运行

启动报错，没有显示是jdk版本问题。但是机器需要jdk8环境要保持旧的ci工具运行。
jenkins依赖jdk11，启动方式为systemctl

# 编辑jenkins.service文件
vi /usr/lib/systemd/system/jenkins.service
# 找到ExecStart这一项
ExecStart=/usr/bin/jenkins
# 编辑/usr/bin/jenkins
vi /usr/bin/jenkins
# 找到java环境变量
infer_java_cmd() {
    JAVA_HOME=/etc/alternatives/java_sdk_11_openjdk   # 这一行是新增的
    if [ -n "${JENKINS_JAVA_CMD}" ] && [ -x "${JENKINS_JAVA_CMD}" ]; then
        return 0
    fi
    if [ -n "${JAVA_HOME}" ] && [ -x "${JAVA_HOME}/bin/java" ]; then
        JENKINS_JAVA_CMD="${JAVA_HOME}/bin/java"
        return 0
    fi
    JENKINS_JAVA_CMD="$(command -v java)" || return "$?"
}
# 重启jenkins即可
# ------ 割了 ------
# 编辑初始化文件
vi /etc/init.d/jenkins
# 找到JENKINS_JAVA_CMD，修改candidates
candidates="
/usr/lib/jvm/java-11-openjdk-11.0.19.0.7-1.0.1.an8.x86_64/bin/java
"
for candidate in $candidates; do
    [ -x "$JENKINS_JAVA_CMD" ] && break
    JENKINS_JAVA_CMD="$candidate"
done
# 注意这里修改完并没有用，所以才直接修改的/usr/bin/jenkins
# 并且/usr/bin/jenkins文件中并没有找到引用/etc/init.d/jenkins的行

启动

systemctl start jenkins
# 初始密码，记住密码浏览器要用。
cat /var/lib/jenkins/secrets/initialAdminPassword
2da3ae75b3af4f69a0d5a48993dbaf8e

nginx代理

# 内网代理
[root@nginx conf.d]# cat jenkins.conf
server {
    listen       80;
    server_name  jenkins.domain.com;
    access_log /home/app/logs/jenkins_access.log  main;
    error_log  /home/app/logs/jenkins_error.log;

    location / {
        proxy_pass   http://192.168.1.17:8080/;
        proxy_set_header Host      $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        # iframe 重定向
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Port $server_port;
        client_max_body_size  50000m;
        client_body_buffer_size 128k;
        #	websocket 配置
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}
# 公网二次代理
[root@localhost wujiops]# cat jenkins.conf
server {
    listen       80;
    server_name  jenkins.domain.com;
    rewrite ^(.*)$  https://$host$1 permanent;
}
server {
    listen 443 ssl;
    server_name jenkins.domain.com;
    ssl_certificate /etc/nginx/cert/domain.com.cer;
    ssl_certificate_key /etc/nginx/cert/domain.com.key;
    ssl_session_timeout 5m;
    ssl_ciphers EECDH+CHACHA20:EECDH+CHACHA20-draft:EECDH+AES128:RSA+AES128:EECDH+AES256:RSA+AES256::!MD5;
    ssl_protocols TLSv1.1 TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;
    add_header Strict-Transport-Security "max-age=31536000";
    access_log /home/logs/jenkins_access.log  main;
    error_log  /home/logs/jenkins_error.log;
    location / {
        proxy_pass   http://10.0.0.18/;
        proxy_set_header Host      $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Port $server_port;
        client_max_body_size  50000m;
        client_body_buffer_size 128k;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}
