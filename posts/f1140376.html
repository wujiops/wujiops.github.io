关闭、卸载 sendmail/postfix

# sendmial
service sendmail stop
chkconfig sendmail off
# postfix
service postfix stop
chkconfig postfix off
# 直接卸载
yum remove sendmail
yum remove postfix

安装mailx

yum install mailx -y

Ubuntu安装：

sudo apt-get install heirloom-mailx

# 时间同步
yum -y install ntpdate
ntpdate 210.72.145.44
ntpdate ntp.api.bz
ntpdate -u 210.72.145.44   # 可以越过主机防火墙
clock -w   # 将时间同步到硬件时钟

---分割线---

# 163邮箱
1)vim /etc/mail.rc   # 最后插入已下内容：
set from=150xxxxxxxx@163.com
set smtp=smtp.163.com
set smtp-auth-user=150xxxxxxxx@163.com
set smtp-auth-password=xxxxxx   # 此密码为授权码，在163网页邮箱可设置
set smtp-auth=login
2)echo "test mail" | mail -vvv -s "subject" xxx@qq.com   # 命令行验证

# qq邮箱
1)vim /etc/mail.rc   # 最后插入已下内容：
set from=xxxx@qq.com
set smtp-auth-user=xxxx@qq.com
set smtp-auth-password=xxxxxx   # 此密码为授权码，在qq邮箱网页版中设置 
set smtp-auth=login
set ssl-verify=ignore
set nss-config-dir=~/.certs
set smtp=smtps://smtp.qq.com:465
2)mkdir ~/.certs && certutil -N -d ~/.certs   # 命令行执行此命令，输入8位密码(任意)，例如Az123123
3)echo "test mail" | mail -vvv -s "subject" 150xxxxxxxx@163.com   # 命令行验证

# 腾讯企业邮箱
1)vim /etc/mail.rc
set from=dev@xxx.com
set smtp-auth-user=dev@xxx.com
set smtp-auth-password=xxxxxx
set smtp-auth=login
set smtp=smtp://smtp.exmail.qq.com
3)echo "test mail" | mail -vvv -s "subject" 150xxxxxxxx@163.com   # 命令行验证
