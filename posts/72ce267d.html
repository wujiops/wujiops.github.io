安装部署

mkdir /home/zentao

podman run -d \
--name zentao \
-v /home/zentao:/data \
-p 180:80 \
-p 13306:3306 \
-e MYSQL_INTERNAL=true \
hub.zentao.net/app/zentao:latest

备份

# 1.backup < 10G
# 登录禅道 --> 左侧菜单项【后台】--> 设置模块的【系统设置】项 --> 顶部【备份】菜单

# 2.backup > 10G
# 进入容器执行命令
docker exec -it zentao bash
cd /apps/zentao/bin
bash init.sh

# 交互命令
# Please input your php path:(example: /usr/bin/php)
# /opt/zbox/bin/php
# Please input zentao url:(example: http://localhost:88/zentao or http://localhost)
# http://localhost
# 生成的脚本
# ztcli.sh ok
# backup.sh ok
# ......

# 执行备份脚本
bash backup.sh

# 查看备份文件
ls -l /data/zentao/tmp/backup/

清理

# 停止禅道容器
podman stop zentao

# 清理停止的容器
podman rm zentao

# 清理旧的镜像
podman images|grep '<none>'|awk '{print $3}'|xargs podman rmi {}
