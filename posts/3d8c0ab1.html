CentOS8重启Network新的配置不生效

开始涉及到以下命令不生效

nmcli c reload ens160
systemctl restart network

网卡配置如下

TYPE="Ethernet"
PROXY_METHOD="none"
BROWSER_ONLY="no"
BOOTPROTO="none"
DEFROUTE="yes"
IPV4_FAILURE_FATAL="yes"
IPV6INIT="yes"
IPV6_AUTOCONF="yes"
IPV6_DEFROUTE="yes"
IPV6_FAILURE_FATAL="no"
IPV6_ADDR_GEN_MODE="stable-privacy"
NAME="ens160"
UUID="1f22ea44-481f-410e-b062-7b05fa7a8375"
DEVICE="ens160"
ONBOOT="yes"
HWADDR="00:0C:29:D5:D6:DE"
IPADDR="192.168.1.223"
PREFIX="24"
GATEWAY="192.168.1.1"
DNS1="114.114.114.114"
IPV6_PRIVACY="no"

解决 reload 配置不生效BUG

# 测试可用
ifdown ens160 && ifup ens160
# 测试可用
nmcli c reload && nmcli d reapply ens160
# 测试可用
nmcli c reload && nmcli networking off && nmcli networking on

总结：以上设置完IP后使用上面的命令可以使新配置生效，涉及到down和off等，一定要一条命令行执行！

养成好习惯，否则只能去服务器执行up和on了。

yum -y install ntpdate

  1. 配置epel源，阿里云Base源（测试不行）

客户端连接超时自动断开，并不会向TCP server发送断开请求，导致server端一直处于连接状态，客户端重连失败。

分析原因

  1. 复现问题

用tcpdump抓包tcpdump tcp port 10000（结果展示如下：）

可以看到客户端106.202.127.124主动断开TCP serverwww.tcpserver.com是正常的（三次握手四次断开）

当客户端是因为超时断开（处于注二状态，TCPserver未得到回复），或者断网（处于注一状态，未发生挥手）

  1. 解决问题

在TCP server设置socket 30s超时，客户端发送消息后，30s没有动作TCP server处于等待连接的状态

这时候需要客户端主动断开重连即可

开启端口，绑定到主机

tcp_update.py

# TCP server
from socket import *
import pymysql
import time
import _thread
import socket

socket.setdefaulttimeout(30)

# Manual Data
db_2 = pymysql.connect(host='$IP', port=$PORT,
                       user='$NAME', passwd='$PASSWORD', db="$DATABASE")
cursor_2 = db_2.cursor()

def process(string: str):
    temp_str = string.split()
    return temp_str[1], float(temp_str[2])

def database_update(attribute, value):
    sql = "UPDATE dataTestUDP SET value = " + \
        str(value) + " WHERE attribute = \"" + attribute + "\""

    try:
        db_2.ping(reconnect=True)
        cursor_2.execute(sql)
        db_2.commit()
    except:
        # 如果发生错误则回滚
        print('Database Error')
        db_2.rollback()
        # 关闭数据库连接
        db_2.close()

def tcp_listener():
    HOST = '192.168.1.2'
    PORT = 10000
    BUFSIZ = 1024
    ADDRESS = (HOST, PORT)

    tcpServerSocket = socket(AF_INET, SOCK_STREAM)

    tcpServerSocket.bind(ADDRESS)
    print("Server starts，listening port {}...".format(ADDRESS[1]))

    tcpServerSocket.listen(5)

    try:
        while True:
            client_socket = None
            try:
                print('Server is running. Waiting client...')
                client_socket, client_address = tcpServerSocket.accept()
                print('Client {} Connected！'.format(client_address))

                while True:
                    try:
                        data = client_socket.recv(2048)
                        if data:
                            print(
                                'Received Message {}({} bytes) from {}'.format(data.decode('gb2312'), len(data),
                                                                               client_address))
                            try:
                                a, v = process(data.decode('gb2312'))
                                database_update(a, v)
                            except:
                                print('Data Error')
                        else:
                            print('Client {} disconnects!'.format(client_address))
                            break
                    except:
                        print('Connenction Reset by {}!'.format(client_address))
                        break
            except:
                print('Timeout.')
                pass
            finally:
                if client_socket is not None:
                    client_socket.close()
    finally:
        tcpServerSocket.close()

if __name__ == '__main__':
    tcp_listener()

# auto update
import pymysql
import time

# Auto Data
db_1 = pymysql.connect(host='$IP', port=$PORT,
                       user='$NAME', passwd='$PASSWORD', db="$DATABASE")
cursor_1 = db_1.cursor()

# Update fake data automatically
def database_auto_update():
    sql_1 = """UPDATE dataTestFake SET id = 0 WHERE id = 10"""
    sql_2 = """UPDATE dataTestFake SET id = id + 1 WHERE id < 10"""

    while True:
        try:
            # 执行sql语句
            db_1.ping(reconnect=True)
            cursor_1.execute(sql_1)
            cursor_1.execute(sql_2)

            # 提交到数据库执行
            db_1.commit()
        except:
            # 如果发生错误则回滚
            db_1.rollback()
            # 关闭数据库连接
            db_1.close()
        time.sleep(4)

if __name__ == '__main__':
    database_auto_update()
